//creation
CREATE TABLE users(
    ... username text,
    ... profilePic text,
    ... country text,
    ... totalUpvotes int,
    ... totalDownvotes int,
    ... creationDate timestamp,
    ... PRIMARY KEY (username));

CREATE TABLE content(
    ... content_id uuid PRIMARY KEY,
    ... title text,
    ... body text,
    ... link text);

CREATE TABLE posts(
    ... post_id uuid PRIMARY KEY,
    ... time timestamp,
    ... subreddit text,
    ... username text,
    ... content_id uuid,
    ... upvotes int,
    ... downvotes int,
    ... labels list<text >);
